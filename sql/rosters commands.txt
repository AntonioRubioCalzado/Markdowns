  SCHOOL_FULL_NAME varchar(30), 
  SCHOOL_NAME varchar(30), 
  NAME varchar(30),
  POSITION varchar(2),
  HEIGHT integer,
  WEIGHT integer,
  YEAR varchar(2),
  HOME_TOWN varchar(30),
  STATE varchar(2),
  ID integer PRIMARY KEY
