##   using the package animation
library(animation)
xs <- seq(-2, 2, len = 100)
oopt <- ani.options(interval = 0.2, nmax = 5, autobrowse = FALSE)
for (i in 1:ani.options("nmax")) {
    plot(dnorm(xs, 0, 1/i), type = "l")
    ani.pause()
}
ani.options(oopt)
